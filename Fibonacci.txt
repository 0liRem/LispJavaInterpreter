;Realizar producción del término n de Factorial de un número

;crear función
(defun factorial (n)

; cálculo para el factorial
;si n es 1 devuelve 1, caso contrario multiplica n por el factorial n - 1
  (if (= n 1)
       1
      (* n (factorial (- n 1)))
))

; escribir el factorial del número
(format t "Factorial: ~a" (factorial 4))
